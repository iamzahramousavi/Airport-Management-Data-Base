CREATE PROCEDURE IU_Airline(carrier_ VARCHAR(2), name_ VARCHAR(50)) BEGIN INSERT INTO Airline(Carrier,Name) VALUES (carrier_, name_); END 
CREATE PROCEDURE IU_Airplane(tailnum_ VARCHAR(6), year_ INTEGER, manufacturer_ VARCHAR(50), Model_ VARCHAR(50), seats_ INTEGER, Airline_ VARCHAR(2)) BEGIN INSERT INTO Airplane(Tailnum,Year,Manufacturer,Model,Seats,Airline) VALUES (tailnum_, year_, manufacturer_, Model_, seats_, Airline_, FOREIGN_); END 
CREATE PROCEDURE IU_Airport(Code_ VARCHAR(3), Name_ VARCHAR(80), Lat_ NUMERIC(15, 8), Lon_ NUMERIC(15, 8), City_ INTEGER) BEGIN INSERT INTO Airport(Code,Name,Lat,Lon,City) VALUES (Code_, Name_, Lat_, Lon_, City_); END 
CREATE PROCEDURE IU_City(Name_ VARCHAR(50), Country_ VARCHAR(20), Population_ INTEGER, Area_ NUMERIC(20,2)) BEGIN INSERT INTO City(Name,Country,Population,Area) VALUES (ID_, Name_, Country_, Population_, Area_); END 
CREATE PROCEDURE IU_Flight(Date_ DATE, FlightNumber_ INTEGER, SchedArrTime_ INTEGER, SchedDepTime_ INTEGER, DepTime_ INTEGER, ArrTime_ INTEGER, Distance_ INTEGER, Airline_ VARCHAR(2), Airplane_ VARCHAR(6)) BEGIN INSERT INTO Flight(ID,Date,FlightNumber,SchedArrTime,SchedDepTime,DepTime,ArrTime,Distance,Airline,Airplane) VALUES (Date_, FlightNumber_, SchedArrTime_, SchedDepTime_, DepTime_, ArrTime_, Distance_, Airline_, Airplane_); END 
CREATE PROCEDURE IU_FlightAttendant(FAID_ INTEGER, YearsOfService_ INTEGER, Salary_ INTEGER, Rank_ VARCHAR(10)) BEGIN INSERT INTO FlightAttendant(FAID,YearsOfService,Salary,Rank) VALUES (FAID_, YearsOfService_, Salary_, Rank_); END 
CREATE PROCEDURE IU_Passenger(PassengerID_ INTEGER, FlyerStatus_ VARCHAR(10)) BEGIN INSERT INTO Passenger(PassengerID,FlyerStatus) VALUES (PassengerID_, FlyerStatus_); END 
CREATE PROCEDURE IU_Person(FirstName_ VARCHAR(20), LastName_ VARCHAR(20), Sex_ VARCHAR(10), Address_ VARCHAR(50), DateOfBirth_ DATE, Occupation_ VARCHAR(50), City_ VARCHAR(50)) BEGIN INSERT INTO Person(FirstName,LastName,Sex,Address,DateOfBirth,Occupation,City) VALUES (FirstName_, LastName_, Sex_, Address_, DateOfBirth_, Occupation_, City_); END 
CREATE PROCEDURE IU_Phone(ID_ INTEGER, Type_ varchar(20), Number_ INTEGER) BEGIN INSERT INTO Phone(ID,Type,Number) VALUES (ID, Type_, Number_); END 
CREATE PROCEDURE IU_Pilot(PilotID_ INTEGER, YearsOfService_ INTEGER, Salary_ INTEGER, Airline_ VARCHAR(2)) BEGIN INSERT INTO Pilot(PilotID,YearsOfService,Salary,Airline) VALUES (PilotID_, YearsOfService_, Salary_, Airline_); END 
CREATE PROCEDURE IU_Route(Origin_ VARCHAR(3), Dest_ VARCHAR(3)) BEGIN INSERT INTO Route(Origin,Dest) VALUES (Origin_, Dest_); END 
CREATE PROCEDURE IU_RouteServe(Date_ DATE, FlightNumber_ INTEGER, SchedArrTime_ INTEGER, SchedDepTime_ INTEGER, Origin_ VARCHAR(3), Dest_ VARCHAR(3)) BEGIN insert into RouteServe (Date, FlightNumber, SchedArrTime, SchedDepTime, Origin, Dest) VALUES (ID_, Date_, FlightNumber_, SchedArrTime_, SchedDepTime_, Origin_, Dest_); END 
CREATE PROCEDURE IU_Take(Date VARCHAR(10), FlightNumber INT, SchedArrTime INT, SchedDepTime INT, PassengerID INT, Type VARCHAR(20), Class VARCHAR(50)) BEGIN INSERT INTO Route(Date, FlightNumber, SchedArrTime, SchedDepTime, PassengerID, Type, Class) VALUES (Date_, FlightNumber_, SchedArrTime_, SchedDepTime_, PassengerID_, Type_, Class_); END
